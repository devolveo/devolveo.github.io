// ✅ Weather API function using PITER methodology
async function fetchWeatherData(city) {
  console.log("API-001: Weather fetch initiated for:", city);
  
  // P - PURPOSE: Get current weather data with caching and error handling
  
  // I - INPUT: Validate city parameter
  if (!city || typeof city !== 'string' || city.trim().length === 0) {
    console.error("API-001 Error: Invalid city parameter");
    return { success: false, error: 'Invalid city name', data: null };
  }
  
  const cleanCity = city.trim();
  const cacheKey = `weather_${cleanCity.toLowerCase()}`;
  
  // Check cache first
  const cached = getCachedWeather(cacheKey);
  if (cached.isValid) {
    console.log("API-001: Using cached data for", cleanCity);
    return { success: true, data: cached.data, source: 'cache' };
  }
  
  // T - TRANSFORM: Fetch from API with proper error handling
  try {
    const response = await fetch(
      `https://api.openweathermap.org/data/2.5/weather?q=${cleanCity}&appid=${API_KEY}&units=metric`
    );
    
    if (!response.ok) {
      throw new Error(`API Error: ${response.status} - ${response.statusText}`);
    }
    
    const weatherData = await response.json();
    
    // Validate API response structure
    if (!weatherData.main || !weatherData.weather) {
      throw new Error('Invalid API response structure');
    }
    
    // Cache successful response
    cacheWeatherData(cacheKey, weatherData);
    console.log("API-001: Fresh data fetched and cached for", cleanCity);
    
    // R - RETURN: Success with data
    return { success: true, data: weatherData, source: 'api' };
    
  } catch (error) {
    // E - ERROR: Handle various failure scenarios
    console.error("API-001 Error:", error.message);
    
    // Try to use stale cache as fallback
    const staleCache = getCachedWeather(cacheKey, true);
    if (staleCache.data) {
      console.warn("API-001: Using stale cache due to API failure");
      return { 
        success: true, 
        data: staleCache.data, 
        source: 'stale_cache',
        warning: 'Data may be outdated due to connectivity issues'
      };
    }
    
    // R - RETURN: Failure with context
    return { 
      success: false, 
      error: error.message, 
      data: null,
      suggestion: 'Check internet connection and try again'
    };
  }
}

PITER Methodology in API Development

The weather application showcases systematic API integration using the PITER framework, ensuring robust and maintainable code for external data sources.

P

Purpose Definition

"What does this API function do?" - Clearly define each function's responsibility: fetch weather, cache data, handle errors, or render UI components with specific, measurable outcomes.

I

Input Validation

"What parameters are needed? What can go wrong?" - Validate city names, API keys, network availability, and handle malformed user input before making API calls.

T

Transform & Process

"How do we get from input to output?" - Design the API call flow: check cache → fetch data → validate response → process for UI display with clear transformation steps.

E

Error Scenarios

"What happens when the API fails?" - Handle network errors, invalid responses, rate limiting, API downtime, and malformed data with graceful fallbacks and user feedback.

R

Return Strategy

"What does the caller need to know?" - Return structured objects with success status, data, error messages, and metadata for debugging and user experience decisions.

Smart Caching Implementation

// Professional caching with expiration logic
function getCachedWeather(cacheKey, allowStale = false) {
  try {
    const cached = localStorage.getItem(cacheKey);
    if (!cached) {
      return { isValid: false, data: null };
    }
    
    const parsedCache = JSON.parse(cached);
    const now = Date.now();
    const cacheAge = now - parsedCache.timestamp;
    
    // Cache valid for 10 minutes
    const CACHE_DURATION = 10 * 60 * 1000;
    
    if (cacheAge < CACHE_DURATION) {
      return { isValid: true, data: parsedCache.data };
    }
    
    // Return stale data if allowed (offline scenarios)
    if (allowStale) {
      return { isValid: false, data: parsedCache.data };
    }
    
    return { isValid: false, data: null };
    
  } catch (error) {
    console.error("Cache read error:", error);
    return { isValid: false, data: null };
  }
}

function cacheWeatherData(cacheKey, data) {
  try {
    const cacheObject = {
      data: data,
      timestamp: Date.now()
    };
    localStorage.setItem(cacheKey, JSON.stringify(cacheObject));
    console.log("Data cached successfully for", cacheKey);
  } catch (error) {
    console.error("Cache write error:", error);
    // App continues to work without caching
  }
}
